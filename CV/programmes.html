                def encode_cesar(s, decalage):
                    chaine_code=""
                    for element in s:
                        if "A"<= element<= "Z":
                            indice=ord(element)-ord("A")
                            indice=indice+decalage
                            chaine_code=chaine_code+chr(indice+ord("A"))
                        else:
                            chaine_code=chaine_code+element
                    return chaine_code

                c_code=encode_cesar("VOICI MON CODE",3)
                print(c_code)
                c_clair=encode_cesar(c_code, -3)
                print(c_clair)
